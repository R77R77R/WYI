module Server.Kestrel

open System
open System.IO
open System.Buffers
open Microsoft.AspNetCore.Builder
open Microsoft.AspNetCore.Http
open Microsoft.Extensions.Hosting
open Microsoft.Extensions.Logging
open System.Net.WebSockets
open Microsoft.Extensions.DependencyInjection



(*

用F#写一个可以在ubuntu上运行的跨平台的HTTP服务，
1. 作为echo服务器，可以接受二进制的HTTP请求，返回二进制HTTP响应。
2. 要求高性能，能支持大量并发ACCEPT
3. 可以扩展为websocket
4. 可以拦截某些类型的URL，实现自己接管的文件服务
5. 支持大文件HTTP上传，断点续传


*)


let builder = WebApplication.CreateBuilder()



// [需求2] 高性能配置：调整 Kestrel 限制以应对大量并发
builder.WebHost.ConfigureKestrel(fun options ->
    options.Limits.MaxConcurrentConnections <- 5000L
    // 允许大文件上传（设为 null 表示不限制单次请求体大小，我们靠流处理）
    options.Limits.MaxRequestBodySize <- Nullable() 
) |> ignore

builder.Services.AddWebSockets(fun _ -> ()) |> ignore
let app = builder.Build()

app.UseWebSockets()

// 设置本地文件存储目录 (确保该目录在 Ubuntu 上已创建并有权限)
let storagePath = "/Dev/storage"
if not (Directory.Exists(storagePath)) then Directory.CreateDirectory(storagePath) |> ignore

// ==========================================
// [需求1 & 需求3] 二进制 Echo & WebSocket 扩展
// ==========================================
app.MapPost("/echo", fun (ctx: HttpContext) ->
    task {
        if ctx.WebSockets.IsWebSocketRequest then
            let! webSocket = ctx.WebSockets.AcceptWebSocketAsync()
            // 使用内存池租用 Buffer，避免高并发下频繁分配内存导致 GC 压力
            let buffer = ArrayPool<byte>.Shared.Rent(4096)
            try
                let mutable result = await (webSocket.ReceiveAsync(ArraySegment(buffer), Threading.CancellationToken.None))
                while not result.CloseStatus.HasValue do
                    // 二进制双工 Echo
                    do! webSocket.SendAsync(ArraySegment(buffer, 0, result.Count), WebSocketMessageType.Binary, result.EndOfMessage, Threading.CancellationToken.None)
                    let! nextResult = webSocket.ReceiveAsync(ArraySegment(buffer), Threading.CancellationToken.None)
                    result <- nextResult
            finally
                ArrayPool<byte>.Shared.Return(buffer)
            return Results.Empty
        else
            // 普通 HTTP 二进制 Echo
            ctx.Response.ContentType <- "application/octet-stream"
            // 直接流复制，极低内存占用
            do! ctx.Request.Body.CopyToAsync(ctx.Response.Body)
            return Results.Empty
    }) |> ignore

// ==========================================
// [需求4 & 需求5 (下)] 接管文件服务 & 支持断点续传下载
// ==========================================
// 拦截 /files/{文件名} 的 GET 请求
app.MapGet("/files/{fileName}", fun (fileName: string) ->
    let filePath = Path.Combine(storagePath, fileName)
    if File.Exists(filePath) then
        // Results.File 原生支持解析 HTTP "Range: bytes=0-1023" 头，自动实现断点续传下载！
        Results.File(filePath, contentType = "application/octet-stream", enableRangeProcessing = true)
    else
        Results.NotFound()
) |> ignore

// ==========================================
// [需求5 (上)] 支持大文件断点续传上传
// ==========================================
// 客户端通过 PUT 请求发送分片，并携带 Content-Range 头，例如：Content-Range: bytes 0-1023/2048
app.MapPut("/upload/{fileName}", fun (fileName: string, ctx: HttpContext) ->
    task {
        let filePath = Path.Combine(storagePath, fileName)
        let rangeHeader = ctx.Request.Headers.["Content-Range"].ToString()
        
        // 解析 Range，这里做简单演示。生产环境建议使用更严谨的正则或系统解析
        let mutable startOffset = 0L
        if not (String.IsNullOrEmpty(rangeHeader)) && rangeHeader.StartsWith("bytes ") then
            let rangePart = rangeHeader.Substring(6).Split('/').[0] // 提取 "0-1023"
            startOffset <- Int64.Parse(rangePart.Split('-').[0])

        // 使用 OpenOrCreate 模式打开文件流
        // FileShare.ReadWrite 允许并发读取（比如边传边下），但在同一文件并发写时需在业务层加锁
        use fileStream = new FileStream(filePath, FileMode.OpenOrCreate, FileAccess.Write, FileShare.ReadWrite)
        
        // 定位到断点位置
        fileStream.Seek(startOffset, SeekOrigin.Begin) |> ignore
        
        // 将请求体流式追加到文件，不占用服务器内存
        do! ctx.Request.Body.CopyToAsync(fileStream)
        
        return Results.Ok("Chunk uploaded")
    }) |> ignore

app.Run("http://0.0.0.0:5000")
